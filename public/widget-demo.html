📝 集成代码

简单集成

将以下代码添加到您的网站 HTML 中即可快速集成智能客服：

<!-- 1. 引入 Widget 脚本 --> <script src="/versions/widget/widget.js"></script> <!-- 2. 初始化 Widget --> <script> CustomerServiceWidget.init({ apiKey: 'your-api-key', // 从管理后台获取 apiEndpoint: 'http://your-domain.com/api/v1', primaryColor: '#4F46E5', position: 'right', headerText: '智能客服', welcomeMessage: '您好！有什么可以帮助您的？' }); </script>

📚 Widget API

方法列表

// 初始化 Widget CustomerServiceWidget.init({ apiKey: 'your-api-key', apiEndpoint: 'http://your-api-endpoint', primaryColor: '#4F46E5', theme: 'light', position: 'right', headerText: '智能客服', welcomeMessage: '您好！', autoOpen: false }); // 打开 Widget CustomerServiceWidget.open(); // 关闭 Widget CustomerServiceWidget.close(); // 发送消息 CustomerServiceWidget.sendMessage('我想退款'); // 更新配置 CustomerServiceWidget.updateConfig({ headerText: '新标题', primaryColor: '#10B981' }); // 设置用户信息 CustomerServiceWidget.setUserInfo({ userId: 'user123', userName: '张三', userEmail: 'zhangsan@example.com' }); // 监听事件 CustomerServiceWidget.on('open', () => { console.log('Widget 已打开'); }); CustomerServiceWidget.on('message:received', (data) => { console.log('收到消息:', data.message); }); CustomerServiceWidget.on('close', () => { console.log('Widget 已关闭'); });
